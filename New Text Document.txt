<script>
  const PRICE = 2000;

  function updatePreview(){
    const qty = document.getElementById('qty').value;
    const color = document.getElementById('roseColor').value;
    const wrap = document.getElementById('wrapColor').value;

    // اسم الصورة حسب الاختيارات
    const fileName = `images/rose-${color}-wrap-${wrap}-${qty}.jpg`;

    // تغيير الصورة
    document.getElementById('bouquetImg').src = fileName;

    // تحديث الملخص
    const total = qty * PRICE;
    const colorText = document.getElementById('roseColor').options[document.getElementById('roseColor').selectedIndex].text;
    const wrapText = document.getElementById('wrapColor').options[document.getElementById('wrapColor').selectedIndex].text;
    document.getElementById('summaryText').textContent =
      `${qty} وردة - ${colorText} - تغليف ${wrapText} - الإجمالي: ${total.toLocaleString('ar-IQ')} د.ع`;
  }
</script>
